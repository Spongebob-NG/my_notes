```
#include<iostream>
#include<stdio.h>
#include<vector>
#include<queue>
using namespace std;
int n, m, x, y;
int a[405][405];
int dx[8] = {1,1,2,2,-1,-1,-2,-2};
int dy[8] = { -2,2,-1,1,-2,2,-1,1 };
struct node {
	int x, y;
};
int main() {
	cin >> n >> m>>x>>y;
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= m; j++) {
			a[i][j] = -1;
		}
	}
	queue<node>q;
	a[x][y] = 0;
	q.push({ x,y });
	while (!q.empty()) {
		node s = q.front();
		q.pop();
		for (int i = 0;i<8; i++) {
				int xi = s.x + dx[i];
				int yi = s.y + dy[i];
				if (xi<1 || xi>n || yi<1 || yi>m)continue;
				if (a[xi][yi] == -1) {
					a[xi][yi] = a[s.x][s.y] + 1;
					q.push({ xi,yi });
			}
		}
	}
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= m; j++) {
			cout << a[i][j] << ' ';
		}
		cout << endl;
	}
}
```

**核心细节：构造了一个结构体，放x和y（横纵坐标），然后构建队列的时候直接把整个结构体放入**
