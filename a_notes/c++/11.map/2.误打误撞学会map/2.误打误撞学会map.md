```
class Solution {
public:
	vector<vector<int>> levelOrder(TreeNode* root) {
		if (!root)return vector<vector<int>>();
		vector<vector<int>>ans;
		map<TreeNode*, int>a;
		queue<TreeNode*>q;
		q.push(root);
		a[root] = 0;
		ans.push_back({ root->val });
		ans.push_back(vector<int>());
		while (!q.empty()) {
			if (q.front()->left) {
				q.push(q.front()->left);
				a[q.front()->left] = a[q.front()] + 1;
				ans[a[q.front()->left]].push_back(q.front()->left->val);
			}
			if (q.front()->right) {
				q.push(q.front()->right);
				a[q.front()->right] = a[q.front()] + 1;
				ans[a[q.front()->right]].push_back(q.front()->right->val);
			}
			if (q.front()->left || q.front()->right)ans.push_back(vector<int>());
			q.pop();
		}
		int n = ans.size()-1;
		while (ans[n].size() == 0) {
			ans.pop_back();
			n--;
		}
		return ans;

	}
}; 
```

这是一个层序遍历二叉树的题，原本很简单的，但是题目要求是返回一个二维数组。

遇到的问题：怎么让遍历的值精准地放在每一行数组中？

解决方案：用map存每个top的层数，然后按层放数据  但是最后一层会有个空行  所以就要在最后再把空行去掉
